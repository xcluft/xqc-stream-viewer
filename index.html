function toggleChatPosition() { if (window.innerWidth > 1000) { if (body.style.flexDirection == 'row-reverse') { body.style.flexDirection = 'row'; setCookie('chatPosition', 'left', 365); } else { body.style.flexDirection = 'row-reverse'; setCookie('chatPosition', 'right', 365); } } } window.addEventListener('resize', function() { if (window.innerWidth <= 1000) { body.style.flexDirection = 'column'; } else { body.style.flexDirection = (getCookie('chatPosition') === 'right') ? 'row-reverse' : 'row'; } }); function setCookie(cname, cvalue, exdays) { var d = new Date(); d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000)); var expires = "expires="+d.toUTCString(); document.cookie = cname + "=" + cvalue + ";" + expires + ";path=/"; } function getCookie(cname) { var name = cname + "="; var ca = document.cookie.split(';'); for(var i = 0; i < ca.length; i++) { var c = ca[i]; while (c.charAt(0) == ' ') { c = c.substring(1); } if (c.indexOf(name) == 0) { return c.substring(name.length, c.length); } } return ""; }
